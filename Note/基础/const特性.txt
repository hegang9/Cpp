#include <iostream>
#include <string>
using namespace std;

//6、常量类成员变量
class MyClass1 {
public:
	const int ID = 1;            // 常量成员（必须在构造函数初始化列表中初始化）
	static const int MAX1 = 100;  // 静态常量（可在类内初始化）

	int value;
	mutable int cache;           // 特殊：可在const成员函数中修改
};

int main()
{
	//1、修饰内置基本类型变量，对于基本类型，const写在类型名前后意义相同
	int const MAX = 100;
	const int MIN = 0;
	cout << MAX << endl;

	//2、指向常量的指针，const在*号之前，指针指向的对象可变，指向的对象的值不可变
	int x=1;
	const int* p1 = &x;     // 类型前 const
	int const* p2 = &x;     // 类型后 const (与上等价)

	//3、常量指针，const在*号之后，指针指向的对象不可变（绑定死了），指向的对象的值可变
	int* const p3 = &x;

	//4、指向常量数据的常量指针，指针和指向的对象的值都不可变
	const int* const p4 = &x;

	//5、常成员函数，const在函数声明末尾
	class MyClass {
	public:
		void normalFunc();                // 普通成员函数
		void constFunc() const;           // const成员函数（位置在参数列表后）
	};


	return 0;
}
