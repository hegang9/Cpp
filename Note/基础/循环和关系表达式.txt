#include <iostream>
#include <string>
using namespace std;


int main()
{
	int i = 0;
	/*while (i++ < 10) {
		cout << i << endl;
	}*/

	int y = (4 + ++i) + (6 + i++);
	cout << y << endl;

	/*
	自增自减运算符的前缀和后缀形式区别：
	1、前缀返回的是左值，可以对齐进行赋值；后缀返回的是右值
	2、前缀先返回自增或自减后的值在参与运算，后缀先参与运算再自增或自减
	3、前缀不会生成临时对象，直接修改元对象；后缀需创建临时对象保存旧值，导致额外构造和析构成本
	4、与指针的解除引用运算符之间的优先级不同：
		（1）前缀递增、前缀递减和解除引用运算符的优先级相同，以从右到左的方式进行结合。
		（2）后缀递增和后缀递减的优先级相同，但比前缀运算符的优先级高，这两个运算符以从左到右的方式进行结合。
		示例：*++p == *(++p) , ++*p == ++(*p) , *p++ == *p,p++（先对p解除引用再自增）
	*/

	//基于范围的for循环
	int prices[10] = { 1,2,3,4,5,6,7,8,9,0 };
	for (int& i : prices) {
		cout << i << " ";
	}
	cout << endl;
	return 0;
}
